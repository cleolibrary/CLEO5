{$CLEO .s}
{$INCLUDE_ONCE ../cleo_tester.inc}

script_name '0A94'
test("0A94 (load_and_launch_custom_mission)", tests)
terminate_this_custom_script


function tests
    before_each(@cleanup)
    after_each(@cleanup)
    
    it("should start custom mission from default location", test1)
    it_cs5("should start script from current folder", test2_cs5)
    it_cs4("should start script from CLEO folder", test2_cs4)
    return
    
    
    :cleanup
        fail_current_mission // end any ongoing mission
        wait {time} 0
    return


    function test1
        // use default search path "cleo\"
        $100 = -1
        $101 = -1
        $102 = -1
        0@ = -2
        1@ = -2
        2@ = -2
        load_and_launch_custom_mission {scriptFileName} "cleo_tests/Cleo/testMission" {args} 3.1415 42
        wait {time} 0
        wait {time} 0
        assert_eqf($100, 3.1415)
        assert_eq($101, 42)
        assert_eq($102, 0)
    end


    function test2_cs5
        // ./ prefix resolves to the current directory in CLEO5
        $100 = -1
        $101 = -1
        $102 = -1
        0@ = -2
        1@ = -2
        2@ = -2
        load_and_launch_custom_mission {scriptFileName} "./testMission" {args} 3.1415 42
        wait {time} 0
        wait {time} 0
        assert_eqf($100, 3.1415)
        assert_eq($101, 42)
        assert_eq($102, 0)
    end
    
    function test2_cs4
        // ./ prefix is relative to CLEO directory
        $100 = -1
        $101 = -1
        $102 = -1
        0@ = -2
        1@ = -2
        2@ = -2
        load_and_launch_custom_mission {scriptFileName} "./cleo_tests/cleo/testMission" {args} 3.1415 42
        wait {time} 0
        wait {time} 0
        assert_eqf($100, 3.1415)
        assert_eq($101, 42)
        assert_eq($102, 0)
    end
end
