function assert_log_contains(find_string: string)
    int f
    copy_file {fileName} LOG_FILE {newFileName} LOG_FILE2

    if f = open_file LOG_FILE2 "rb"
    then

        int size = get_file_size f
        int source = allocate_memory size

        if read_block_from_file f size source
        then
            int p = strstr(source, find_string)
            close_file f
            free_memory source
            assert_ptr(p)
        else
            close_file f
            free_memory source
            trace "Error: can't read file %s" LOG_FILE2
            assert(false) // fail
        end
    else
        trace "Error: can't open file %s" LOG_FILE2
        assert(false) // fail
    end

    function strstr<cdecl, 0x822650>(str: string, substr: string): string
end
