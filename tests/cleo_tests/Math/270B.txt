{$CLEO .s}
{$INCLUDE_ONCE ../cleo_tester.inc}

script_name '270B'
test("270B (float_mul)", tests)
terminate_this_custom_script


function tests
    it("should multiply two floats", test1)
    return
    
    function test1
        float a = 10.0
        float b = 40.0
        float c = 80.0
        270B: c = a * b
        assert_eqf(a, 10.0)
        assert_eqf(b, 40.0)
        assert_eqf(c, 400.0)
    end
end
