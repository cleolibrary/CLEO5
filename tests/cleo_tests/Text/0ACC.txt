{$CLEO .s}
{$INCLUDE_ONCE ../cleo_tester.inc}

script_name '0ACC'
test("0ACC (print_string)", tests)
terminate_this_custom_script

function tests
    before_each(@setup)
    after_each(@cleanup)

    it("should print short string variable", test1)
    it("should print long string variable", test2)
    it("should print string pointer", test3)
    it("should print very long string pointer", test4)
    it_cs5("should add CLEO texts to brief history", test5_cs5)
    it_cs4("should not add CLEO texts to brief history", test5_cs4)
    it_cs5("should reset bAddNextMessageToPreviousBriefs flag after print", test6_cs5)
    it_cs4("should not reset bAddNextMessageToPreviousBriefs flag after print", test6_cs4)
    it_cs5("should add 9th message to brief after queue wraps around", test7_queue_wrap)
    it_cs4("should not add 9th message to brief in legacy mode after queue wraps", test7_queue_wrap_cs4)
    it_cs5("should not add duplicate text to brief history", test8_queue_dedup)
    it_cs4("should not add any text to brief history in legacy mode", test8_queue_dedup_cs4)
    return

    :cleanup
    CMessages_ClearPreviousBriefArray()
    return

    :setup
    clear_prints
    return

    function test1
        shortString a = 'From ACC' // 8 characters
        shortString b = '12345678'

        print_string {text} a {time} 100
        a = '--VOID--'
        wait {time} 0

        int str = getCurrSubtitleText()
        assert_eqs(str, "From ACC")
    end

    function test2
        longString a = "Message from ACC" // 16 characters
        shortString b = '12345678'

        print_string {text} a {time} 100
        a = "--VOID--"
        wait {time} 0

        int str = getCurrSubtitleText()
        assert_eqs(str, "Message from ACC")
    end

    function test3
        int msg = allocate_memory {size} 64
        string_format msg = "Hello from 0ACC!"

        print_string {text} msg {time} 100
        string_format msg = "--VOID--"
        wait {time} 0

        free_memory {address} msg
        msg = 0

        int str = getCurrSubtitleText()
        assert_eqs(str, "Hello from 0ACC!")
    end

    function test4
        int msg = allocate_memory {size} 513
        write_memory {address} msg {size} 512 {value} 0x41 {vp} false // fill with A

        print_string {text} msg {time} 100
        string_format msg = "--VOID--"
        wait {time} 0

        free_memory {address} msg
        msg = 0

        int expected = allocate_memory {size} 400 // same as game's internal limit
        write_memory {address} expected {size} 399 {value} 0x41 {vp} false // fill with A

        int str = getCurrSubtitleText()

        // assert_eqs(str, expected) // assert error message will not be able to display strings that long
        is_text_equal {text} str {another} expected {ignoreCase} false
        assert_result_true()

        free_memory {address} expected
        expected = 0
    end

    function test5_cs5
        print_now {key} 'STAT000' {time} 100 {flag} 1 // add Progress made to brief
        print_now {key} 'STAT001' {time} 100 {flag} 1 // add Total progress to brief

        print_brief_messages()

        // only msg1 is added immediately (queue was empty); msg2 is skipped (flag=false); msg3 is deferred (queued at slot 2)
        assert_brief_at(0, "Brief 0ACC one")
        assert_brief_at(1, "Total progress")
        assert_brief_at(2, "Progress made")
    end

    function test5_cs4
        print_now {key} 'STAT013' {time} 100 {flag} 1 // add Weapon Budget to brief
        print_now {key} 'STAT014' {time} 100 {flag} 1 // add Fashion Budget to brief
        print_now {key} 'STAT015' {time} 100 {flag} 1 // add Property Budget to brief

        print_brief_messages()

        assert_brief_at(0, "Property Budget")
        assert_brief_at(1, "Fashion Budget")
        assert_brief_at(2, "Weapon Budget")
    end

    function test6_cs5
        add_next_message_to_previous_briefs {state} false
        print_string {text} "Flag test ACC" {time} 100

        int flag = read_memory {address} 0x00A4448B {size} 1 {vp} false // CTheScripts::bAddNextMessageToPreviousBriefs
        assert_true(flag) // flag must be reset to true after print_string
    end

    function test6_cs4
        add_next_message_to_previous_briefs {state} false
        print_string {text} "Flag test ACC cs4" {time} 100

        int flag = read_memory {address} 0x00A4448B {size} 1 {vp} false // CTheScripts::bAddNextMessageToPreviousBriefs
        assert_false(flag) // flag must NOT be reset for legacy scripts

        add_next_message_to_previous_briefs {state} true // restore flag
    end

    function test7_queue_wrap
        // Fill the entire 8-slot queue with short-lived messages.
        // After wait 1000ms they are long gone; their pointers (msgQueueBuff[0..7]) still live in
        // PreviousBriefs. On the 9th call msgQueueBuff[0] is reused - without the null-out fix
        // AddToPreviousBriefArray would reject the new text as a duplicate pointer.
        print_string {text} "Wrap msg 1" {time} 100
        print_string {text} "Wrap msg 2" {time} 100
        print_string {text} "Wrap msg 3" {time} 100
        print_string {text} "Wrap msg 4" {time} 100
        print_string {text} "Wrap msg 5" {time} 100
        print_string {text} "Wrap msg 6" {time} 100
        print_string {text} "Wrap msg 7" {time} 100
        print_string {text} "Wrap msg 8" {time} 100

        wait {time} 1500 // let all 8 messages expire and be dequeued by the game

        print_string {text} "Wrap msg 9" {time} 100 // reuses msgQueueBuff[0]; must still reach brief

        assert_brief_at(0, "Wrap msg 9")
    end

    function test7_queue_wrap_cs4
        // Same queue-wrap scenario in legacy mode: message must still display but must NOT appear in brief.
        print_string {text} "Wrap msg 1" {time} 100
        print_string {text} "Wrap msg 2" {time} 100
        print_string {text} "Wrap msg 3" {time} 100
        print_string {text} "Wrap msg 4" {time} 100
        print_string {text} "Wrap msg 5" {time} 100
        print_string {text} "Wrap msg 6" {time} 100
        print_string {text} "Wrap msg 7" {time} 100
        print_string {text} "Wrap msg 8" {time} 100

        wait {time} 1500 // let all 8 messages expire and be dequeued by the game

        print_string {text} "Wrap msg 9" {time} 9000 // reuses msgQueueBuff[0]; must display but not reach brief
        wait {time} 0

        int str = getCurrSubtitleText()
        assert_eqs(str, "Wrap msg 9") // message must still display correctly

        int brief = getBriefTextAt(0)
        assert_eq(brief, 0) // brief must remain empty in legacy mode
    end
    function test8_queue_dedup
        // Printing the same text twice must produce only one brief entry.
        // msg1 becomes current immediately (empty queue) -> brief[0] = "Dedup check".
        // msg2 queues behind msg1; after msg1 expires msg2 becomes current and the game
        // would call AddToPreviousBriefArray for it - but only if addToBrief was set.
        // Our content-based dedup in AddToMessageQueue must suppress that flag for msg2.
        print_string {text} "Dedup check" {time} 100 // msg1: becomes current, brief[0] set
        print_string {text} "Dedup check" {time} 100 // msg2: dedup flag cleared at enqueue time

        wait {time} 500 // let both messages expire so msg2 has cycled through to current

        assert_brief_at(0, "Dedup check") // exactly one entry
        int second = getBriefTextAt(1)
        assert_eq(second, 0) // brief[1] must be empty - dedup suppressed the second entry
    end

    function test8_queue_dedup_cs4
        // In legacy mode print_string must never touch the brief regardless of duplicates.
        print_string {text} "Dedup check cs4" {time} 100
        print_string {text} "Dedup check cs4" {time} 100

        wait {time} 500

        int brief = getBriefTextAt(0)
        assert_eq(brief, 0) // brief must remain empty in legacy mode
    end
end

function assert_brief_at(index: int, expected: string)
    int str = getBriefTextAt(index)
    assert_eqs(str, expected)
end

function print_brief_messages
    int msg1 = allocate_memory {size} 64
    int msg2 = allocate_memory {size} 64
    int msg3 = allocate_memory {size} 64
    string_format msg1 = "Brief 0ACC one"
    string_format msg2 = "Brief 0ACC two"
    string_format msg3 = "Brief 0ACC three"

    clear_prints

    print_string {text} msg1 {time} 9000            // msg1 is added to brief

    add_next_message_to_previous_briefs {state} false
    print_string {text} msg2 {time} 9000            // msg2 is NOT added to brief

    add_next_message_to_previous_briefs {state} true    // no effect as flag is true already
    print_string {text} msg3 {time} 9000            // msg3 will be added to brief when it becomes current (deferred)

    string_format msg1 = "--VOID--"
    string_format msg2 = "--VOID--"
    string_format msg3 = "--VOID--"

    free_memory {address} msg1
    free_memory {address} msg2
    free_memory {address} msg3
    msg1 = 0
    msg2 = 0
    msg3 = 0
end

function CMessages_ClearPreviousBriefArray<cdecl, 0x69DE70>()

// get pointer to current subtitle text
function getCurrSubtitleText(): int
    int txt = read_memory {address} 0x00C1A7F0 {size} 4 {vp} false // GTA SA 1.0: CMessages::BriefMessages[0].text
    return txt
end

// get pointer to previous brief
function getBriefTextAt(index: int): int
    int offset = index * 0x20 // sizeof(PreviousBrief) = 0x20, m_pText at offset 0
    int txt = read_memory_with_offset {address} 0x00C1A570 {offset} offset {size} 4 // GTA SA 1.0: CMessages::PreviousBriefs[index].m_pText
    return txt
end
